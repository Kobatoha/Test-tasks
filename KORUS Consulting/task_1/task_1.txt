#Найти города, в которых в 2016 году было издано больше всего книг
SELECT книги.'Город издания', COUNT(*) AS 'Количество книг'
FROM книги
WHERE YEAR(DATE_SUB('Дата поступления в библиотеку', INTERVAL 5 YEAR)) = 2016
GROUP BY книги.'Город издания'
ORDER BY 'Количество книг' DESC;


#Вывести количество экземпляров книг «Война и мир» Л.Н.Толстого, которые находятся в библиотеке
SELECT COUNT(*) AS 'Количество экземпляров'
FROM книги
JOIN выдачи_книг ON книги.'ID экземпляра' = выдачи_книг.'ID экземпляра'
WHERE книги.'Название' = 'Война и мир' AND книги.'Автор' = 'Л.Н.Толстой' AND выдачи_книг.'Дата возврата' IS NULL;


#Найти читателей, которые за последний месяц брали больше всего книг в библиотеке c сортировкой читателей по возрасту (от молодых к старшим)
SELECT читатели.'Фамилия', читатели.'Имя', читатели.'Отчество', читатели.'Дата рождения'
FROM читатели
JOIN выдачи_книг ON читатели.'№ читательского билета' = выдачи_книг.'№ читательского билета'
WHERE выдачи_книг.'Дата выдачи' >= DATE_SUB(NOW(), INTERVAL 1 MONTH)
GROUP BY читатели.'№ читательского билета'
ORDER BY читатели.'Дата рождения' ASC;
